\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\tag{limit}
Consider 
\[
f(w) = \frac{-2}{w+1}.
\]
Compute
\begin{enumerate}
\item $\lim_{w\to -1^- } f(w) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{w\to -1^+ } f(w) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{w\to -1 } f(w) \begin{prompt} = \answer{DNE}\end{prompt}$
\end{enumerate}
\end{exercise}
