\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\tag{limit}
Consider 
\[
f(z) = \frac{-4}{z+4}.
\]
Compute
\begin{enumerate}
\item $\lim_{z\to -4^- } f(z) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{z\to -4^+ } f(z) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{z\to -4 } f(z) \begin{prompt} = \answer{DNE}\end{prompt}$
\end{enumerate}
\end{exercise}
