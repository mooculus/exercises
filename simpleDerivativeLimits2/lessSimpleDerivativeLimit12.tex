\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $y(u) = \sqrt{u+9}$. Compute

\[
\lim_{ u\to -5 } 
\frac{y(u)-y(-5)}{u+5} \begin{prompt}=\answer{\frac{1}{4}}\end{prompt}
\]
\end{exercise}
