\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $A(t) = \sqrt{t+2}$. Compute

\[
\lim_{ t\to -1 } 
\frac{A(t)-A(-1)}{t+1} \begin{prompt}=\answer{\frac{1}{2}}\end{prompt}
\]
