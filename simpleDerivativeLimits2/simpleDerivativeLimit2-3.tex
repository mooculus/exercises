\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $b(n) = \sqrt{n+5}$. Compute

\[
\lim_{ n\to -3 } 
\frac{b(n)-b(-3)}{n+3} \begin{prompt}=\answer{\frac{1}{2 \sqrt{2}}}\end{prompt}
\]
\end{exercise}
