\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $A(u) = \sqrt{u+1}$. Compute

\[
\lim_{ u\to 1 } 
\frac{A(u)-A(1)}{u-1} \begin{prompt}=\answer{\frac{1}{2 \sqrt{2}}}\end{prompt}
\]
\end{exercise}
