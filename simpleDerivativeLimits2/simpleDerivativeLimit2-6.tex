\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $C(n) = \sqrt{n+10}$. Compute

\[
\lim_{ n\to 5 } 
\frac{C(n)-C(5)}{n-5} \begin{prompt}=\answer{\frac{1}{2 \sqrt{15}}}\end{prompt}
\]
