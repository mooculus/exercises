\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $b(\theta) = \sqrt{\theta +5}$. Compute

\[
\lim_{ \theta\to 4 } 
\frac{b(\theta)-b(4)}{\theta -4} \begin{prompt}=\answer{\frac{1}{6}}\end{prompt}
\]
