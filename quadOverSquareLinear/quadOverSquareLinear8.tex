\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
Consider 
\[
f(\theta) = \frac{\theta ^2-1}{(\theta -2)^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{\theta\to 2^- } f(\theta) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{\theta\to 2^+ } f(\theta) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{\theta\to 2 } f(\theta) \begin{prompt} = \answer{\infty}\end{prompt}$
\end{enumerate}
\end{exercise}
