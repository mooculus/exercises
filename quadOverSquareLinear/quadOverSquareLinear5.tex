\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
Consider 
\[
f(n) = \frac{n^2+4 n+3}{(n+3)^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{n\to -3^- } f(n) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{n\to -3^+ } f(n) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{n\to -3 } f(n) \begin{prompt} = \answer{DNE}\end{prompt}$
\end{enumerate}
\end{exercise}
