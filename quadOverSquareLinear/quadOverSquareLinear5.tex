\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
Consider 
\[
f(w) = \frac{w^2+2 w-8}{(w-4)^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{w\to 4^- } f(w) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{w\to 4^+ } f(w) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{w\to 4 } f(w) \begin{prompt} = \answer{\infty}\end{prompt}$
\end{enumerate}
\end{exercise}
