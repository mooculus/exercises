\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
Consider 
\[
f(v) = \frac{v^2+3 v-4}{(v+3)^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{v\to -3^- } f(v) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{v\to -3^+ } f(v) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{v\to -3 } f(v) \begin{prompt} = \answer{-\infty}\end{prompt}$
\end{enumerate}
\end{exercise}
