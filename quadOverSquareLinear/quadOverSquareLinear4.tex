\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
Consider 
\[
f(u) = \frac{u^2+2 u-3}{(u+2)^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{u\to -2^- } f(u) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{u\to -2^+ } f(u) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{u\to -2 } f(u) \begin{prompt} = \answer{-\infty}\end{prompt}$
\end{enumerate}
\end{exercise}
