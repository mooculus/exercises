\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\tag{limit}
Consider 
\[
f(k) = \frac{-4}{(k+3)^4}.
\]
Compute
\begin{enumerate}
\item $\lim_{k\to -3^- } f(k) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{k\to -3^+ } f(k) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{k\to -3 } f(k) \begin{prompt} = \answer{-\infty}\end{prompt}$
\end{enumerate}
\end{exercise}
