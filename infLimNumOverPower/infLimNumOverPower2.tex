\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\tag{limit}
Consider 
\[
f(t) = \frac{-4}{(t+4)^3}.
\]
Compute
\begin{enumerate}
\item $\lim_{t\to -4^- } f(t) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{t\to -4^+ } f(t) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{t\to -4 } f(t) \begin{prompt} = \answer{DNE}\end{prompt}$
\end{enumerate}
\end{exercise}
