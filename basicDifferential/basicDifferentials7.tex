\begin{exercise}
\outcome{Compute differentials.}

\tag{derivative}
Let $y= \sqrt[3]{3} \sqrt[3]{x} \csc (2 x)$. Compute the differential $dy$.

\[
\begin{prompt}dy =\answer{(\frac{\csc (2 x)}{3^{2/3} \sqrt[3]{x}^2}-2 \sqrt[3]{3} \sqrt[3]{x} \cot (2 x) \csc (2 x)) dx}\end{prompt}
\]
