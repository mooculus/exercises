\begin{exercise}
\outcome{Compute differentials.}

\tag{derivative}
Let $y= \frac{e^{-x}}{\sqrt{2} \sqrt{x}}$. Compute the differential $dy$.

\[
\begin{prompt}dy =\answer{(-\frac{e^{-x}}{2 \sqrt{2} x^{3/2}}-\frac{e^{-x}}{\sqrt{2} \sqrt{x}}) dx}\end{prompt}
\]
