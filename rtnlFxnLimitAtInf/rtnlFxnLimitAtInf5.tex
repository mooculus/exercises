\begin{exercise}
\outcome{Calculate the limit as $x$ approaches $\pm \infty$ of common functions algebraically.}
\outcome{Compute limits at infinity of famous functions.}
\tag{limit}
Let 
\[
B(u) = \frac{-3 u^3-3 u^2+4}{-3 u^3-3 u^2+4}.
\]
Compute
\begin{enumerate}
\item $\lim_{u\to \infty} B(u) \begin{prompt} = \answer{1}\end{prompt}$
\item $\lim_{u\to -\infty}B(u) \begin{prompt} = \answer{1}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{u^3}}{\frac{1}{u^3}}
\]
\end{hint}
\end{exercise}
