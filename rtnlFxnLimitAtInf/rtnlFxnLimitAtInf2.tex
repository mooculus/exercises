\begin{exercise}
\outcome{Calculate the limit as $x$ approaches $\pm \infty$ of common functions algebraically.}
\outcome{Compute limits at infinity of famous functions.}
\tag{limit}
Let 
\[
F(z) = \frac{-5 z^3-3}{z^3-5 z^2-2 z}.
\]
Compute
\begin{enumerate}
\item $\lim_{z\to \infty} F(z) \begin{prompt} = \answer{-5}\end{prompt}$
\item $\lim_{z\to -\infty}F(z) \begin{prompt} = \answer{-5}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{z^3}}{\frac{1}{z^3}}
\]
\end{hint}
\end{exercise}
