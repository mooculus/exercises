\begin{exercise}
\outcome{Calculate the limit as $x$ approaches $\pm \infty$ of common functions algebraically.}
\outcome{Compute limits at infinity of famous functions.}
\tag{limit}
Let 
\[
a(k) = \frac{3 k^2+k-5}{4 k^2+2 k+2}.
\]
Compute
\begin{enumerate}
\item $\lim_{k\to \infty} a(k) \begin{prompt} = \answer{\frac{3}{4}}\end{prompt}$
\item $\lim_{k\to -\infty}a(k) \begin{prompt} = \answer{\frac{3}{4}}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{k^2}}{\frac{1}{k^2}}
\]
\end{hint}
\end{exercise}
