\begin{exercise}

\outcome{Answer questions about lines and curves in space.}

\tag{curves}


Let $\vecl_{1}(t)$ be the line whose components are $\left\{ \begin{aligned}x(t) & =1.1+0.6t\\
y(t) & =3.77+0.9t\\
z(t) & =-2.3+1.5t
\end{aligned}
\right.$  and $\vecl_{2}(t)$ be the line whose components are $\left\{ \begin{aligned}x(t) & =3.11+3.4t\\
y(t) & =2+5.1t\\
z(t) & =2.5+8.5t
\end{aligned}
\right.$

Then $\vecl_{1}$ and $\vecl_{2}$ are: 

\begin{multipleChoice}
\choice[correct]{parallel}
\choice{intersecting}
\choice{non-intersecting}
\choice{the same}
\end{multipleChoice}


\end{exercise}
