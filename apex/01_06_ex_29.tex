\begin{exercise}

\outcome{Calculate limits of piecewise functions.}
\outcome{Identify where a function is and is not continuous.}

\tag{piecewise}
\tag{continuity}

Let
\[
f(x) =
\begin{cases}
  x^2-1 &x < 3 \\
  x+5   &x\geq 3.
\end{cases}
\]
Is $f$ continuous everywhere? 

\begin{multipleChoice}
\choice[correct]{Yes}
\choice{No}
\end{multipleChoice}

\end{exercise}
