\begin{exercise}

\outcome{Compute cross products.}

\tag{cross products}
\tag{vectors}

Let $\vec{u}=\vector{1,0,1}$ and $\vec{v}=\vector{5,0,7}$. Compute the cross product $\vec{u} \cross \vec{v}$.

\begin{prompt}
\[
\vec{u}\cross\vec{v}=\vector{\answer{0},\answer{-2},\answer{0}}
\]
\end{prompt}
