\outcome{Explain the relationship between one-sided and two-sided limits.}

\tag{limit}

\begin{quote}
If $\lim_{x\to1^{-}}f(x)=5$, then $\lim_{x\to1}f(x)=5$.
\end{quote}

\begin{prompt}
\begin{multipleChoice}
\choice{True}
\choice[correct]{False}
\end{multipleChoice}
\end{prompt}
