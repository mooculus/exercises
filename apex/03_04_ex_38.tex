\begin{exercise}

\outcome{Find all local maximums and minimums using the 1st and 2nd derivative tests.}
\outcome{Classify critical points.}


\tag{local extrema}
\tag{derivative}
\tag{derivative test}
\tag{first derivative test}
\tag{second derivative test}



Let $f(x)=\sin(x)+\cos(x)$.
\begin{enumerate}
\item		$f$ has local maxima on $(-\pi,\pi)$ at $x=\answer{\pi/4}$, write DNE if no such point(s) exist.
\item		$f$ has local minima on $(-\pi,\pi)$ at $x=\answer{-3\pi/4}$, write DNE if no such point(s) exist.
\end{enumerate}
