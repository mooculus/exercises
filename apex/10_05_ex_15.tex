\begin{exercise}

\outcome{Answer questions about lines and curves in space.}

\tag{curves}


Let $\vecl_{1}(t)=\vector{1,2,1}+t\vector{2,-1,1}$ and $\vecl_{2}(t)=\vector{3,3,3}+t\vector{-4,2,-2}$. 

Then $\vecl_{1}$ and $\vecl_{2}$ are: 

\begin{multipleChoice}
\choice[correct]{parallel}
\choice{intersecting}
\choice{non-intersecting}
\choice{the same}
\end{multipleChoice}

\end{exercise}
