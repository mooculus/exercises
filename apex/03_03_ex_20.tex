\outcome{Classify critical points.}





\tag{increasing}
\tag{decreasing}
\tag{derivative test}

Let $f(x)=\frac{x}{x^2-2x-8}$. 
\begin{enumerate}
\item		The domain of $f$ is $\answer{(-\infty,-2)\cup(-2,4)\cup(4,\infty)}$.
\item		$f$ has a critical point(s) at $x = \answer{DNE}$, write DNE if no such point(s) exist.
\item		$f$ is decreasing on $\answer{(-\infty,-2)\cup(-2,4)\cup(4,\infty)}$, write DNE if no such interval(s)exist.
\item		$f$ is increasing on $\answer{DNE}$, write DNE if no such interval(s) exist.
\item		At $x=0$, $f$ has 
\begin{multipleChoice}
\choice{A local minimum}
\choice{A local maximum}
\choice{Both a local maximum and a local minimum}
\choice[correct]{Neither a local maximum nor a local minimum}
\end{multipleChoice}
\end{enumerate}
