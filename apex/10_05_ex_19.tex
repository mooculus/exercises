\begin{exercise}

\outcome{Answer questions about lines and curves in space.}

\tag{curves}


Let $\vecl_{1}(t)$ be the line whose components are $\left\{ \begin{aligned}x(t) & =1+2t\\
y(t) & =3-2t\\
z(t) & =t
\end{aligned}
\right.$  and $\vecl_{2}(t)$ be the line whose components are $\left\{ \begin{aligned}x(t) & =3-t\\
y(t) & =3+5t\\
z(t) & =2+7t
\end{aligned}
\right.$

Then $\vecl_{1}$ and $\vecl_{2}$ are: 

\begin{multipleChoice}
\choice{parallel}
\choice{intersecting}
\choice[correct]{non-intersecting}
\choice{the same}
\end{multipleChoice}


\end{exercise}
