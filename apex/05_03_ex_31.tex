\begin{exercise}

\outcome{Add up a large number of terms quickly using sigma notation.}
\outcome{Compute left, right, and midpoint Riemann Sums with 10 or fewer rectangles.}
\outcome{Understand the relationship between area under a curve and sums of rectangles.}
\outcome{Approximate area under a curve.}
\outcome{Understand how the area under a curve is related to the antiderivative.}

\tag{sum}
\tag{Riemann sum}
\tag{integral}

Using $3$ equally spaced rectangles, compute the midpoint Riemann sum approximating $\int_{1}^{2}\ln(x)\d x$. Write your answer as a decimal, rounded to the nearest thousandth. 
\[
\int_{1}^{2}\ln(x)\d x\approx\answer{0.389}
\]

\end{exercise}
