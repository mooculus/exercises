\begin{exercise}

\outcome{Compute Taylor polynomials.}

\tag{power series}
\tag{taylor polynomial}


Find the Taylor polynomial of degree $5$ of $f(x)=\frac{1}[x}$ at $c=2$.

\[
\begin{prompt}
p_{5}(x)=\answer{\frac{1}{2}-\frac{x-2}{4}+\frac{1}{8} (x-2)^2-\frac{1}{16}
   (x-2)^3+\frac{1}{32} (x-2)^4-\frac{1}{64} (x-2)^5}
\end{prompt}
\]

\end{exercise}
