\begin{exercise}
\outcome{Calculate limits of the form zero over zero.}
\tag{limit}
\tag{indeterminate form}
Let $r(w) = \sqrt{w+3}$. Compute

\[
\lim_{ w\to -1 } 
\frac{r(w)-r(-1)}{w+1} \begin{prompt}=\answer{\frac{1}{2 \sqrt{2}}}\end{prompt}
\]
\end{exercise}
