\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{w\to4}Y(w)=1$, $\lim_{w\to4}f(w)=-4$, and $\lim_{w\to4}P(w)=1$. Compute the limit

\[
\lim_{w\to 4 } \frac{Y(w)}{f(w)-P(w)}\begin{prompt} = \answer{-\frac{1}{5}}\end{prompt}
\]
\end{exercise}
