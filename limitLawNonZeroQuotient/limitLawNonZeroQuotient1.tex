\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{t\to4}b(t)=-1$, $\lim_{t\to4}r(t)=2$, and $\lim_{t\to4}R(t)=3$. Compute the limit

\[
\lim_{t\to 4 } \frac{b(t)}{r(t)-R(t)}\begin{prompt} = \answer{1}\end{prompt}
\]
\end{exercise}
