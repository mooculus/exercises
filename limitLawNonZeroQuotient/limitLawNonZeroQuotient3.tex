\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{v\to5}b(v)=-2$, $\lim_{v\to5}B(v)=-4$, and $\lim_{v\to5}R(v)=-3$. Compute the limit

\[
\lim_{v\to 5 } \frac{b(v)}{B(v)-R(v)}\begin{prompt} = \answer{2}\end{prompt}
\]
\end{exercise}
