\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{n\to-3}F(n)=-2$, $\lim_{n\to-3}B(n)=4$, and $\lim_{n\to-3}g(n)=3$. Compute the limit

\[
\lim_{n\to -3 } \frac{F(n)}{B(n)-g(n)}\begin{prompt} = \answer{-2}\end{prompt}
\]
\end{exercise}
