\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{z\to-2}f(z)=1$ and $\lim_{z\to-2}C(z)=5$. Compute the limit

\[
\lim_{z\to -2 } \sqrt[6]{f(z)C(z)+5}\begin{prompt} = \answer{0}\end{prompt}
\]
\end{exercise}
