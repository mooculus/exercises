\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{v\to3}r(v)=1$ and $\lim_{v\to3}f(v)=2$. Compute the limit

\[
\lim_{v\to 3 } \sqrt[7]{r(v)f(v)+2}\begin{prompt} = \answer{\sqrt[7]{3}}\end{prompt}
\]
\end{exercise}
