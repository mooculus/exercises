\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{z\to5}Y(z)=1$ and $\lim_{z\to5}C(z)=-2$. Compute the limit

\[
\lim_{z\to 5 } \sqrt[5]{Y(z)C(z)-2}\begin{prompt} = \answer{0}\end{prompt}
\]
\end{exercise}
