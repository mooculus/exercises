\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{w\to5}C(w)=4$ and $\lim_{w\to5}Y(w)=4$. Compute the limit

\[
\lim_{w\to 5 } \sqrt[5]{C(w)Y(w)+4}\begin{prompt} = \answer{\sqrt[5]{2} 3^{2/5}}\end{prompt}
\]
\end{exercise}
