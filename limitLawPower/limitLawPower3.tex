\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{\psi\to-2}y(\psi)=1$ and $\lim_{\psi\to-2}a(\psi)=2$. Compute the limit

\[
\lim_{\psi\to -2 } \sqrt[5]{y(\psi)a(\psi)+2}\begin{prompt} = \answer{2^{2/5}}\end{prompt}
\]
\end{exercise}
