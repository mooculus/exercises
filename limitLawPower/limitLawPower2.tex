\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{limit laws}
Suppose that $\lim_{k\to-5}F(k)=1$ and $\lim_{k\to-5}b(k)=5$. Compute the limit

\[
\lim_{k\to -5 } \sqrt[4]{F(k)b(k)+5}\begin{prompt} = \answer{\sqrt{2}}\end{prompt}
\]
\end{exercise}
