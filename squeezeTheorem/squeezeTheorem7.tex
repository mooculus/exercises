\begin{exercise}

\outcome{Understand the Squeeze Theorem and how it can be used to find limit values.}
\outcome{Calculate limits using the Squeeze Theorem.}
\tag{limit}
\tag{squeeze theorem}

	Let $M(t)$ be a function which satisfies the inequalities:
	
	\[
		5 \leq \frac{M(t)}{t^2} \leq  \frac{t^2+t-6}{t-2} 
	\]
	
	whenever $t \geq 2$.
	
	Then 
	
	\[
	\lim_{t \to 2^+} M(t)  =  \answer{20}
	\]
	
	\begin{hint}
		First manipulate the inequality so that $M$ is squeezed between two functions.  Then use the squeeze theorem.
	\end{hint}

\end{exercise}
